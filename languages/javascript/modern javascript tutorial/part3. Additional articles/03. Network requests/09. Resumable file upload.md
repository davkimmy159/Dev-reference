파일 업로드 재개
====

##### `fetch` <sub>(메서드)</sub>
- 파일 업로드 용이

##### 업로드 재개
- 내장 기능 無
- 다수 기능 부분적 구현 가능

##### 용량 큰 파일 업로드 재개 작업
- 업로드 진행률 정보 필요
  - `fetch` <sub>(메서드)</sub> 불가능
  - `XMLHttpRequest` <sub>(객체)</sub> 사용

### 무익한 업로드 진행 이벤트

#### `upload.onprogress`

##### 업로드 진행률 추적
- 데이터 전송 중 주기적 발생

##### 서버 측 데이터 수신 여부 확인 불가능 <sub>(브라우저)</sub>
- 다양한 문제 발생 가능
  - 지역 네트워크 프락시 지연
  - 원격 서버 프로세스 문제 발생
  - 전송 중 손실 발생 · 리시버 도달 X
  - 기타 등등

##### 업로드 재개 시 필요 정보
- 정확한 byte 숫자 <sub>(서버 데이터)</sub>
  - 추가 요청 필요

### 알고리즘

##### 1. 파일 ID 생성 <sub>(파일 고윳값)</sub>
- 업로드 재개 대상 파일 서버 인식
- 이름 · 크기 · 최종 수정 일시 변경 시
  - 별도 `fileId` <sub>(변수)</sub> 생성
```javascript
let fileId =
    file.name
  + '-'
  + file.size
  + '-'
  + +file.lastModifiedDate;
```

##### 2. 요청 전송
- 이전 byte 전송량 질의
- 서버 측 파일 업로드 추적 작업 구현 가정
  - `X-File-Id` <sub>(헤더)</sub>
- 서버 내 파일 부재 시
  - `0` 응답 전송
```javascript
let response = await fetch('status', {
  headers: {
    'X-File-Id': fileId
  }
});

// 서버 측 파일 수신량 (byte 숫자) 확인
let startByte = +await response.text();
```

##### 3. 파일 전송
- `startByte` <sub>(변수)</sub>
  - 업로드 재개 시작 위치 (byte)
- `slice` <sub>(`Blob` 객체 내 메서드)</sub> 사용
```javascript
xhr.open("POST", "upload", true);

// 파일 ID
// - 업로드 대상 파일 서버 인식
xhr.setRequestHeader('X-File-Id', fileId);

// 시작 byte (업로드 재개 위치)
// - 서버 측 파일 업로드 재개 인지
xhr.setRequestHeader('X-Start-Byte', startByte);

xhr.upload.onprogress = (e) => {
  `Uploaded ${startByte + e.loaded} of ${startByte + e.total}`;
};

// 업로드 대상 파일
// - ex) input.files[0] 등
xhr.send(file.slice(startByte));
```

#### 요청 헤더

##### `X-File-Id` <sub>(헤더 · 파일 ID)</sub>
- 업로드 대상 파일 서버 인식

##### `X-Start-Byte` <sub>(헤더 · 시작 byte)</sub>
- 업로드 초기화 X · 재개 서버 인식

#### 서버 작업

##### 기록 확인
- 파일 업로드 유무 여부
- 현재 서버 내 업로드 된 파일 크기 확인
  - `X-Start-Byte` <sub>(헤더)</sub> 값 일치 여부

##### 파일 데이터 추가
- 서버 내 업로드 된 파일 크기 시점부터 시작

```html
<!-- index.html -->
<!DOCTYPE HTML>
<html>

<head>
  <script src="uploader.js"></script>
</head>

<body>
  <form name="upload" method="POST" enctype="multipart/form-data" action="/upload">
    <input type="file" name="myfile">
    <input type="submit" name="submit" value="Upload (Resumes automatically)">
  </form>

  <button onclick="uploader.stop()">Stop upload</button>
  <div id="log">Progress indication</div>

  <script>
    function log(html) {
      document.getElementById('log').innerHTML = html;

      html;
    }

    function onProgress(loaded, total) {
      log("progress " + loaded + ' / ' + total);
    }

    let uploader;

    document.forms.upload.onsubmit = async function(e) {
      e.preventDefault();

      let file = this.elements.myfile.files[0];
      if (!file) {
        return;
      }

      uploader = new Uploader({file, onProgress});

      try {
        let uploaded = await uploader.upload();

        if (uploaded) {
          log('success');
        } else {
          log('stopped');
        }
      } catch(err) {
        err;

        log('error');
      }
    };
  </script>
</body>

</html>
```

```javascript
/* server.js */
let http = require('http');
let static = require('node-static');
let fileServer = new static.Server('.');
let path = require('path');
let fs = require('fs');
let debug = require('debug')('example:resume-upload');

let uploads = Object.create(null);

function onUpload(req, res) {

  let fileId = req.headers['x-file-id'];
  let startByte = +req.headers['x-start-byte'];

  if (!fileId) {
    res.writeHead(400, "No file id");
    res.end();
  }

  // 파일 저장 위치 지정
//let filePath = path.join('/tmp', fileId);
  let filePath = '/dev/null';

  debug("onUpload fileId: ", fileId);

  // 새 업로드 초기화
  if (!uploads[fileId]) {
    uploads[fileId] = {};
  }

  let upload = uploads[fileId];

  debug("bytesReceived:" + upload.bytesReceived + " startByte:" + startByte)

  let fileStream;

  /* startByte 값
   0 · 無
   - 새 파일 생성
   有
   - 크기 확인 · 기존 파일 이어 진행
   */
  if (!startByte) {
    upload.bytesReceived = 0;
    fileStream = fs.createWriteStream(filePath, {
      flags: 'w'
    });

    debug("New file created: " + filePath);
  } else {

    // 디스크 내 파일 크기 추가 확인 가능
    if (upload.bytesReceived != startByte) {
      res.writeHead(400, "Wrong start byte");
      res.end(upload.bytesReceived);
      return;
    }

    // 기존 파일 이어 진행
    fileStream = fs.createWriteStream(filePath, {
      flags: 'a'
    });

    debug("File reopened: " + filePath);
  }

  req.on('data', function(data) {
    debug("bytes received", upload.bytesReceived);

    upload.bytesReceived += data.length;
  });

  // 요청 본문 → 파일
  req.pipe(fileStream);

  // 요청 작업 · 요청 데이터 쓰기 완료
  fileStream.on('close', function() {
    if (upload.bytesReceived == req.headers['x-file-size']) {
      debug("Upload finished");

      delete uploads[fileId];

      /* 업로드 완료 파일 대상 추가 작업 진행 */

      res.end("Success " + upload.bytesReceived);
    } else {

      // 연결 종료 · 업로드 미완료 파일 방치
      debug("File unfinished, stopped at " + upload.bytesReceived);
      res.end();
    }
  });

  // I/O 에러 시 요청 작업 종료
  fileStream.on('error', function(err) {
    debug("fileStream error");

    res.writeHead(500, "File error");
    res.end();
  });

}

function onStatus(req, res) {
  let fileId = req.headers['x-file-id'];
  let upload = uploads[fileId];

  debug("onStatus fileId:", fileId, " upload:", upload);

  if (!upload) {
    res.end("0")
  } else {
    res.end(String(upload.bytesReceived));
  }
}

function accept(req, res) {
  if (req.url == '/status') {
    onStatus(req, res);
  } else if (req.url == '/upload' && req.method == 'POST') {
    onUpload(req, res);
  } else {
    fileServer.serve(req, res);
  }
}

// -----------------------------------

if (!module.parent) {
  http.createServer(accept).listen(8080);

  'Server listening at port 8080';
} else {
  exports.accept = accept;
}
```
```javascript
/* uploader.js */
class Uploader {
  constructor({file, onProgress}) {
    this.file = file;
    this.onProgress = onProgress;

    // fileId (파일 ID) 생성
    // - 파일 고윳값
    // - 사용자 세션 고윳값 추가 가능
    this.fileId =
      file.name +
      '-' +
      file.size +
      '-' +
      +file.lastModifiedDate;
  }

  async getUploadedBytes() {
    let response = await fetch('status', {
      headers: {
        'X-File-Id': this.fileId
      }
    });

    if (response.status != 200) {
      throw new Error("Can't get uploaded bytes: " + response.statusText);
    }

    let text = await response.text();

    return +text;
  }

  async upload() {
    this.startByte = await this.getUploadedBytes();

    let xhr = this.xhr = new XMLHttpRequest();

    xhr.open("POST", "upload", true);

    // 파일 ID
    // - 업로드 재개 파일 서버 인지
    xhr.setRequestHeader('X-File-Id', this.fileId);

    // 업로드 재개 시작 위치 (byte)
    // - 업로드 재개 동작 서버 인지
    xhr.setRequestHeader('X-Start-Byte', this.startByte);

    xhr.upload.onprogress = (e) => {
      this.onProgress(this.startByte + e.loaded, this.startByte + e.total);
    };

    `send the file, starting from: ${this.startByte}`;

    xhr.send(this.file.slice(this.startByte));

    /*
     업로드 성공
     - true 반환
     업로드 취소
     - false 반환
     에러
     - 예외 던지기
     */
    return await new Promise((resolve, reject) => {

      xhr.onload = xhr.onerror = () => {
        `upload end status: ${xhr.status}, text: ${xhr.statusText}`;

        if (xhr.status == 200) {
          resolve(true);
        } else {
          reject(new Error("Upload failed: " + xhr.statusText));
        }
      };

      // abort() (업로드 취소 메서드) 호출 시 실행
      xhr.onabort = () => resolve(false);
    });
  }

  stop() {
    if (this.xhr) {
      this.xhr.abort();
    }
  }
}
```
