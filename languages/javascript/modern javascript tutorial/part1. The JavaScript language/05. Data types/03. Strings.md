##### `\n` <sub>(줄 바꿈 문자)</sub>
- `"` · `'` <sub>(큰 · 작은따옴표)</sub> 내 다수 줄 삽입 가능
```javascript
let guestList = "손님:\n * John\n * Pete\n * Mary";

/*
"손님:
 * John
 * Pete
 * Mary"
 */
alert(guestList);
```

##### 따옴표 내 `\n` <sub>(줄 바꿈)</sub> vs `` ` `` <sub>(역따옴표 · 백틱)</sub> 내 다수 줄
- 동일 형태 <sub>(표현 방식만 상이)</sub>
```javascript
// `\n` (줄 바꿈) 사용
let str1 = "Hello\nWorld";

// '`' (역따옴표 · 백틱) 사용
let str2 = `Hello
World`;

// 동일 문자열
alert(str1 == str2); // true
```

#### `String.indexOf(substr, pos)`
- 문자열의 `pos` <sub>(위치)</sub> 에서 시작
- `substr` <sub>(부분 문자열)</sub> 위치 검색
- 부분 문자열 찾음
  - 위치 반환
- 부분 문자열 못 찾음

```javascript
let str = 'Widget with id';

//  0 ('Widget' with id)
alert( str.indexOf('Widget') );

// -1 (대 · 소문자 구분)
alert( str.indexOf('widget') );

//  1 (W'id'get with id)
```

##### `if` <sub>(문)</sub> 조건식
- `0` <sub>(문자열 1번째 위치)</sub> == `false`
- `-1` 과 비교
```javascript
let str = "Widget with id";

  alert("찾았다!");
}

// -1 과 비교
// 의도한 대로 동작
if (str.indexOf("Widget") != -1) {
    alert("찾았다!");

```

#### `~` <sub>(비트 NOT 연산자)</sub> 사용 기법

##### 오래전부터 전해 오는 방법
1. 피연산자 32비트 정수 변환 <sub>(소수부 모두 버려짐)</sub>
2. 모든 비트 반전
    - `-n` → `-(n+1)`
```javascript
alert( ~2 );  // -3 → -( 2+1)
alert( ~1 );  // -2 → -( 1+1)
alert( ~0 );  // -1 → -( 0+1)
alert( ~-1 ); //  0 → -(-1+1)
```

##### `~-1` → `n == 0`
- 非직관적 <sub>(지양)</sub>
```javascript
let str = "Widget";
if (~str.indexOf("Widget")) {
  alert( '찾았다!' );
}
```

##### 아주 긴 문자열 주의
- `~4294967295` == `0`

#### `includes` · `[start·end]sWith`
- 비교적 근래 등장 메서드

##### `String.includes(substr, pos)`
- 문자열에 `substr` <sub>(부분 문자열)</sub>
  - 존재 시
    - `true` 반환
  - 부재 시
    - `false` 반환
- 포함 여부만 확인
  - 위치 정보 X
- 2번째 인수 <sub>(`pos`)</sub>
  - 검색 시작 위치
```javascript
// true
alert( "Widget with id".includes("Widget") );

// false
alert( "Hello".includes("Bye") );

// true
alert( "Widget".includes("id") );

// false (3번째 위치 이후 "id" 없음)
alert( "Widget".includes("id", 3) );
```

##### `String.[start·end]sWith`
- 문자열 특정 문자열 시작 · 종료 여부 확인
```javascript
// true
alert( "Widget".startsWith("Wid") );

// true
alert( "Widget".endsWith("get") );
```

#### `String.slice(start [, end])`
- 특정 문자열 위치 반환

```javascript

```

##### 2번째 인수 <sub>(end)</sub> 생략 시
- 범위 : `start` ~ 문자열 끝
```javascript
let str = "stringify";
                     // stringify
                     // 012345678

```

##### `start` · `end` : 음수 가능
- 문자열 끝에서 카운팅 시작
```javascript
let str = "stringify";

//     ngif
// stringify
//     ↑ ~ ↑
// 0  -5  -1

```

#### `String.substring(start [, end])`
- 특정 문자열 위치 반환
  - `start` ~ (`end` - 1)
- `slice` 와의 차이
  - `start` \> `end` 가능
  - 음수 인수 X
    - `0` 처리
```javascript
let str = "stringify";
                              //       012345678
alert( str.substring(2, 6) ); // ring, stringify
                              //         s   e
alert( str.substring(6, 2) ); // ring, stringify
                              //         e   s

                              //       012345678
alert( str.slice(2, 6) );     // ring, stringify
                              //         s   e
alert( str.slice(6, 2) );     // ""  , stringify
                              //             s
```

#### `String.substr(start [, length])`
- `start` 부터 `length` 개 글자 반환
- 1번째 인수 : 음수
  - 뒤에서 카운트
```javascript
let str = "stringify";
alert( str.substr(2, 4) );  // ring, stringify
                            //       0 2
                            //         s---

let str = "stringify";
alert( str.substr(-4, 2) ); // gi  , stringify
                            //       0   -4
                            //            s-
```
- `slice(start [, end])`
  - `start` ~ (`end` - 1)
  - 음수 허용
- `substring(start [, end])`
  - `start` ~ (`end` - 1)
  - 음수 `0` 취급
- `substr(start [, length])`
  - `start` ~ `length` 개
  - 음수 허용

##### 서로게이트 쌍 : 2글자 취급
- 기호 가져오기 까다로움
- 서로게이트 쌍 구성 글자들
  - 붙어있을 때만 의미 有
```javascript
alert( '𝒳'[0] ); // 이상한 기호 출력
alert( '𝒳'[1] ); // 서로게이트 쌍 일부 출력
```

##### 서로게이트 쌍 대응 코드 사용해 감지 가능 <sub>(기술적)</sub>
- `0xd800` ~ `0xdbff` <sub>(글자 코드)</sub>
  - 쌍 구성 1번째 글자
- `0xdc00` ~ `0xdfff` <sub>(글자 코드)</sub>
  - 쌍 구성 2번째 글자
- `0xd800` ~ `0xdbff` · `0xdc00` ~ `0xdfff`
  - 표준 내 서로게이트 쌍 위해 비워둔 코드
```javascript
// charCodeAt
// - 서로게이트 쌍 처리 X
// - 서로게이트 쌍 구성 부분 코드 반환

// d835 (0xd800 ~ 0xdbff)
alert( '𝒳'.charCodeAt(0).toString(16) );

// dcb3 (0xdc00 ~ 0xdfff)
alert( '𝒳'.charCodeAt(1).toString(16) );
```

#### 발음 구별 기호 · 유니코드 정규화

##### 합성 글자 구성
- 베이스 글자 위 · 아래 + 발음 구별 기호
  - ex\) a → àáâäãåā
- 대부분 독자적인 코드 보유 <sub>(UTF-16 테이블)</sub>
- 모든 합성 글자 코드 부여 X
  - 조합 가능 글자 수 너무 많음
- UTF-16 내 몇 개 유니코드 문자 남겨둠
  - 임의 조합 지원
- 베이스 글자 + 1 ~ n개 유니코드 문자
```javascript
// `S` + `\u0307`   // `Ṡ`
alert( 'S\u0307' ); //  Ṡ

// `Ṡ` + `\u0323`         // `Ṩ`
alert( 'S\u0307\u0323' ); //  Ṩ
```

##### 장점
- 엄청난 유연성 제공

##### 단점
- 동일 글자 · 다른 유니코드 조합
```javascript
// S + 윗 점 + 아랫 점
let s1 = 'S\u0307\u0323'; // Ṩ

// S + 아랫 점 + 윗 점
let s2 = 'S\u0323\u0307'; // Ṩ

alert( `s1: ${s1}, s2: ${s2}` );

// 눈에는 동일 글자
// - 동등 비교 시 상이
alert( s1 == s2 ); // false
```

##### 해결
- 유니코드 정규화 알고리즘 사용
- 각 문자열 동일한 형태로 정규화

##### `String.normalize()`
- 유니코드 정규화 알고리즘 구현
```javascript
// true
alert( "S\u0307\u0323".normalize() == "S\u0323\u0307".normalize() );

// 3개 글자 → 1개 글자 (결합)
// Ṩ '\u1e68' (유니코드)

// 1
alert( "S\u0307\u0323".normalize().length );

// true
alert( "S\u0307\u0323".normalize() == "\u1e68" );
```

##### 현실 : UTF-16 작업자들의 주관적인 의견
- `Ṩ` 코드 부여
  - '충분히 나타날 수 있는 사례' 간주

- [Unicode Normalization Forms](https://www.unicode.org/reports/tr15/ "Unicode Normalization Forms")

##### JS 문자열 수정 불가능

##### `""` <sub>(빈 문자열)</sub> 예외
- `str[0]` == `undefined`
  - `toUpperCase()` <sub>(메서드)</sub> X
    - 에러

```javascript
function ucFirst(str) {
```

2. `maxlength` <sub>(최대 길이 인수)</sub> 초과 시
    - 길이 초과분 문자열

```javascript
function truncate(str, maxlength) {

}
```
