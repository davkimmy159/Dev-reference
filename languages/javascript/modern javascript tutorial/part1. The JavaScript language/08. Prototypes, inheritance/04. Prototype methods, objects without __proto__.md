##### `__proto__`
- 브라우저 대상 개발 시

- 표준 관련 내용 명시

#### 모던 메서드

##### `Object.create(proto, [descriptors])`
- `proto` 참조하는 `{}` <sub>(빈 객체)</sub> 생성
  - `[[Prototype]]` → `proto` 참조
- `descriptors` <sub>(프로퍼티 설명자)</sub>
  - 선택적 전달

##### `Object.getPrototypeOf(obj)`
- `obj.[[Prototype]]` 반환

##### `Object.setPrototypeOf(obj, proto)`
- `obj.[[Prototype]]` → `proto` 설정
```javascript
let animal = {
  eats: true
};

// 새 객체 생성
// - 프로토타입 → animal 참조
let rabbit = Object.create(animal);

// true
alert(rabbit.eats);

// true
alert(Object.getPrototypeOf(rabbit) === animal);

// rabbit (자식 객체) 프로토타입 → {} 참조
Object.setPrototypeOf(rabbit, {});
```

##### `Object.create`
- 프로퍼티 설명자
  - 선택적 전달
```javascript
let animal = {
  eats: true
};

let rabbit = Object.create(animal, {
  jumps: {
    value: true
  }
});

alert(rabbit.jumps); // true
```
- `for…in` 보다 더 8효과적으로 객체 복제
```javascript
let clone = Object.create(
  Object.getPrototypeOf(obj),
  Object.getOwnPropertyDescriptors(obj)
);
```

### 사전에 `toString` 추가하기

##### `dictionary` <sub>(객체)</sub>
- `Object.create(null)` 생성
- 키 · 값 쌍 저장

##### `toString` <sub>(메서드)</sub> 추가
- 키 목록 <sub>(`,` 쉼표 구분 문자열)</sub> 반환
- `for…in` <sub>(반복문)</sub> 순회 배제
```javascript
let dictionary = Object.create(null);

/* 메서드 추가 코드 작성 */

// 데이터 추가
dictionary.apple = "Apple";

// __proto__
// - 일반 프로퍼티 키
dictionary.__proto__ = "test";

// apple · __proto__
// - 반복문 순회 대상
for (let key in dictionary) {

  // "apple", "__proto__"
  alert(key);
}

// "apple,__proto__"
// - toString (메서드) 동작 부분
alert(dictionary);
```

##### 열거 가능한 <sub>(`Object.keys`)</sub> 모든 키 목록 출력 가능
- 프로퍼티 설명자 사용해 정의
  - `toString` <sub>(메서드)</sub> 열거 X

##### `Object.create` 문법
- 2번째 인수
  - 프로퍼티 설명자
```javascript
let dictionary = Object.create(null, {

  // toString (프로퍼티) 정의
  toString: {

    // 함수 정의
    value() {
      return Object.keys(this).join();
    }
    // 설명자 사용해 프로퍼티 생성 시 기본 플래그값
    // - `false`
  }
});

dictionary.apple = "Apple";
dictionary.__proto__ = "test";

for (let key in dictionary) {

  // "apple" "__proto__"
  alert(key);
}

// "apple,__proto__"
// - toString
//   - `,` (쉼표) 구분 프로퍼티 목록 문자열
alert(dictionary);
```

### 호출 간의 차이

##### 새 `rabbit` <sub>(객체)</sub> 생성
```javascript
function Rabbit(name) {
  this.name = name;
}
Rabbit.prototype.sayHi = function() {
  alert(this.name);
};

let rabbit = new Rabbit("Rabbit");
```

##### 코드 동작 결과 동일 · 상이 예측
```javascript
rabbit.sayHi();
Rabbit.prototype.sayHi();
Object.getPrototypeOf(rabbit).sayHi();
rabbit.__proto__.sayHi();
```

#### `this` : 실제 점 앞 객체

##### 1번째 호출
- `this`
  - `rabbit`
- `this.name`
  - `"Rabbit"`

##### 다른 호출
- `this`
  - `Rabbit.prototype`
- `this.name`
  - `undefined`
```javascript
function Rabbit(name) {
  this.name = name;
}
Rabbit.prototype.sayHi = function() {
  alert( this.name );
}

let rabbit = new Rabbit("Rabbit");

// Rabbit
rabbit.sayHi();

// undefined
Rabbit.prototype.sayHi();

// undefined
Object.getPrototypeOf(rabbit).sayHi();

// undefined
rabbit.__proto__.sayHi();
```
