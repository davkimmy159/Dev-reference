##### 정적 메서드
- `prototype` 메서드 X
- 클래스 함수
  - 클래스 자체 메서드 설정
- `static` <sub>(키워드)</sub>
```javascript
class User {

  // 정적 메서드
  static staticMethod() {

    // this 값
    // - User (클래스 생성자) 자체
    // - 점 앞 객체

    // true
    alert(this === User);
  }
}

↓↓↓

// 정적 메서드
// - 프로퍼티 직접 할당
class User { }
User.staticMethod = function() {
  alert(this === User); // true
};

// 사용법
User.staticMethod(); // true
```

##### 특정 객체 X
- 클래스 속한 함수
```javascript
class Article {
  constructor(title, date) {
    this.title = title;
    this.date = date;
  }

  static compare(articleA, articleB) {
    return articleA.date - articleB.date;
  }
}

let articles = [
  new Article("HTML", new Date(2019, 1, 1)),
  new Article("CSS", new Date(2019, 0, 1)),
  new Article("JavaScript", new Date(2019, 11, 1))
];

articles.sort(Article.compare);

// CSS
alert( articles[0].title );
```

### 정적 프로퍼티 · 메서드 상속
```javascript
class Animal {
  static planet = "지구";

  constructor(name, speed) {
    this.speed = speed;
    this.name = name;
  }

  run(speed = 0) {
    this.speed += speed;
    alert(`${this.name}가 속도 ${this.speed}로 달립니다.`);
  }

  static compare(animalA, animalB) {
    return animalA.speed - animalB.speed;
  }

}

// Animal 상속
class Rabbit extends Animal {
  hide() {
    alert(`${this.name}가 숨었습니다!`);
  }
}

let rabbits = [
  new Rabbit("흰 토끼", 10),
  new Rabbit("검은 토끼", 5)
];

// 상속된 정적 메서드
rabbits.sort(Rabbit.compare);

// 검은 토끼가 속도 5로 달립니다.
rabbits[0].run();

// 지구
// - 상속된 정적 프로퍼티
alert(Rabbit.planet);
```

##### `Rabbit extends Animal`
- `[[Prototype]]` 참조 2개 생성
  - 둘 다 프로토타입 통해 상속
1. `Rabbit` <sub>(함수)</sub> → `Animal` <sub>(함수)</sub>
2. `Rabbit.prototype` → `Animal.prototype`
```javascript
class Animal {}
class Rabbit extends Animal {}

// true
// - 정적 메서드
alert(Rabbit.__proto__ === Animal);

// true
// - 일반 메서드
alert(Rabbit.prototype.__proto__ === Animal.prototype);
```

### `Object` 상속 클래스

##### 객체 : 보통 `Object.prototype` 상속
- '일반' 객체 메서드 접근 가능

  - 기타 등등

```javascript
class Rabbit {

  }
}

let rabbit = new Rabbit("Rab");

// true

alert( rabbit.hasOwnProperty('name') );
```

##### `Object` 명시적 vs 암묵적 상속 차이
- `class Rabbit extends Object`
- `class Rabbit`

##### 코드 수정

```javascript
class Rabbit extends Object {
```

##### `super()` 미호출
- 상속 클래스 생성자
  - 반드시 호출
  - 미호출 시
    - `this` 정의 X
```javascript
class Rabbit extends Object {
  constructor(name) {
    super();
    this.name = name;
  }
}

let rabbit = new Rabbit("Rab");

// true
alert( rabbit.hasOwnProperty('name') );
```

##### `extends` 문법 : 프로토타입 설정
1. 생성자 함수 `prototype` 사이
    - 일반 메서드용
2. 생성자 함수 자체 사이
    - 정적 메서드용

##### `class Rabbit extends Object`
```javascript
class Rabbit extends Object {}

// 생성자 함수 prototype 사이 (일반 메서드용)
// - true
alert( Rabbit.prototype.__proto__ === Object.prototype );

// 생성자 함수 자체 사이 (정적 메서드용)
// - true
alert( Rabbit.__proto__ === Object );
```

##### `Rabbit` <sub>(클래스)</sub>
- `Object` 정적 메서드 접근 가능
```javascript
class Rabbit extends Object {}

// 원래 호출 방법
// - Object.getOwnPropertyNames
alert ( Rabbit.getOwnPropertyNames({ a: 1, b: 2 })); // a,b
```

##### `extends Object` 부재 시
- `Rabbit.__proto__`
  - `Object` 설정 X
- `Rabbit` <sub>(클래스)</sub>
  - `Object` 정적 메서드 사용 X
```javascript
class Rabbit {}

// 생성자 함수 prototype 사이 (일반 메서드용)
// - true
alert( Rabbit.prototype.__proto__ === Object.prototype );

// 생성자 함수 자체 사이 (정적 메서드용)
// - false
alert( Rabbit.__proto__ === Object );

// 모든 함수 기본 프로토타입
// - true
alert( Rabbit.__proto__ === Function.prototype );

// error, no such function in Rabbit
// - Error
alert ( Rabbit.getOwnPropertyNames({ a: 1, b: 2 }));
```
