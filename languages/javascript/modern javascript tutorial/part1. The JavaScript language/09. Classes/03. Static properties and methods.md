### 정적 메서드

##### `static` <sub>(키워드)</sub>
- 클래스 함수 자체 내 저장
  - `prototype` <sub>(프로토타입)</sub> 내 저장 X 
```javascript
class User {

  // 정적 메서드
  static staticMethod() {

 // this == 클래스 생성자
    this === User; // true
  }
}

↓↓↓

// 프로퍼티 직접 할당 형태
class User {}

User.staticMethod = function() {
  this === User; // true
};

User.staticMethod(); // true
```

##### `static compare(…)` <sub>(정적 메서드)</sub>
- 생성 객체 간 비교 코드 저장
```javascript
class Article {
  constructor(title, date) {
    this.title = title;
    this.date  = date;
  }

  // 정적 메서드 (생성 객체 비교)
  static compare(articleA, articleB) {
    return articleA.date - articleB.date;
  }
}

let articles = [
  new Article("HTML", new Date(2019, 1, 1)),
  new Article("CSS", new Date(2019, 0, 1)),
  new Article("JavaScript", new Date(2019, 11, 1))
];

// 생성 객체 간 비교 시 정적 메서드 사용
articles.sort(Article.compare);

articles[0].title; // CSS
```

##### `static createToday()` <sub>(팩토리 메서드)</sub>
- 특정 조건 부합 객체 생성
  - 매개변수 정보 저장 객체
  - 현재 일자 기반 빈 객체
  - 기타 등등
```javascript
class Article {
  constructor(title, date) {
    this.title = title;
    this.date = date;
  }

  // (정적) 팩토리 메서드
  static createTodays() {

    // this == Article (생성자)
    return new this("Today's digest", new Date());
  }
}

let article = Article.createTodays();

article.title; // Today's digest
```

### 정적 메서드 · 프로퍼티 상속

##### 정적 메서드 · 프로퍼티
- 정상 상속

##### `Rabbit` <sub>(클래스)</sub> → `Animal` <sub>(클래스)</sub> 상속
- `Animal.compare` <sub>(정적 메서드)</sub>
  - `Rabbit.compare`
- `Animal.planet` <sub>(정적 프로퍼티)</sub>
  - `Rabbit.planet`
```javascript
class Animal {

  // 정적 프로퍼티
  static planet = "지구";

  constructor(name, speed) {
    this.speed = speed;
    this.name = name;
  }

  // 정적 메서드
  static compare(animalA, animalB) {
    return animalA.speed - animalB.speed;
  }

  run(speed = 0) {
    this.speed += speed;

    alert(`${this.name}가 속도 ${this.speed}로 달립니다.`);
  }
}

// Rabbit (클래스) → Animal (클래스) 상속
class Rabbit extends Animal {
  hide() {
    alert(`${this.name}가 숨었습니다!`);
  }
}

let rabbits = [
  new Rabbit("흰 토끼", 10),
  new Rabbit("검은 토끼", 5)
];

// 정적 상속 메서드
rabbits.sort(Rabbit.compare);

// 일반 상속 메서드
rabbits[0].run(); // 검은 토끼가 속도 5로 달립니다.


// 정적 상속 프로퍼티
Rabbit.planet; // 지구
```

##### `Rabbit extends Animal`
- `[[Prototype]]` 참조 2개 생성
  - 둘 다 프로토타입 통해 상속
1. `Rabbit` <sub>(함수)</sub> → `Animal` <sub>(함수)</sub>
2. `Rabbit.prototype` → `Animal.prototype`
```javascript
class Animal {}
class Rabbit extends Animal {}

// true
// - 정적 메서드
alert(Rabbit.__proto__ === Animal);

// true
// - 일반 메서드
alert(Rabbit.prototype.__proto__ === Animal.prototype);
```

### `Object` 상속 클래스

##### 객체 : 보통 `Object.prototype` 상속
- '일반' 객체 메서드 접근 가능
  - `hasOwnProperty`
  - 기타 등등
```javascript
class Rabbit {
  constructor(name) {
    this.name = name;
  }
}

let rabbit = new Rabbit("Rab");

// true
// - hasOwnProperty (메서드)
//   - Object.prototype
alert( rabbit.hasOwnProperty('name') );
```

##### `Object` 명시적 vs 암묵적 상속 차이
- `class Rabbit extends Object`
- `class Rabbit`

##### 코드 수정
- 명시적 상속 코드
  - 비정상 동작
```javascript
class Rabbit extends Object {

    this.name = name;
  }
```

##### `super()` 미호출
- 상속 클래스 생성자
  - 호출 필수
  - 미호출 시
    - `this` 정의 X
```javascript
class Rabbit extends Object {
  constructor(name) {
    super();
    this.name = name;
  }
}

let rabbit = new Rabbit("Rab");

// true
alert( rabbit.hasOwnProperty('name') );
```

##### `extends` 문법 : 프로토타입 설정
1. 생성자 `prototype` 사이
    - 일반 메서드용
2. 생성자 자체 사이
    - 정적 메서드용

##### `class Rabbit extends Object`
```javascript
class Rabbit extends Object {}

// 생성자 prototype 사이 (일반 메서드용)
// - true
alert( Rabbit.prototype.__proto__ === Object.prototype );

// 생성자 자체 사이 (정적 메서드용)
// - true
alert( Rabbit.__proto__ === Object );
```

##### `Rabbit` <sub>(클래스)</sub>
- `Object` 정적 메서드 접근 가능
```javascript
class Rabbit extends Object {}

// 기존 호출 방법
// - Object.getOwnPropertyNames
alert ( Rabbit.getOwnPropertyNames({a: 1, b: 2})); // a,b
```

##### `extends Object` 부재 시
- `Rabbit.__proto__`
  - `Object` 설정 X
- `Rabbit` <sub>(클래스)</sub>
  - `Object` 정적 메서드 사용 X
```javascript
class Rabbit {}

// 생성자 prototype 사이 (일반 메서드용)
// - true
alert( Rabbit.prototype.__proto__ === Object.prototype );

// 생성자 자체 사이 (정적 메서드용)
// - false
alert( Rabbit.__proto__ === Object );

// 모든 함수 기본 프로토타입
// - true
alert( Rabbit.__proto__ === Function.prototype );

// error, no such function in Rabbit
// - Error
alert ( Rabbit.getOwnPropertyNames({a: 1, b: 2}));
```
