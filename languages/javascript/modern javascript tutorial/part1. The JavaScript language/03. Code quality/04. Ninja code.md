### 부작용 있는 코드 작성하기

##### 부작용 유무 여부
- 부작용 없는 함수
  - 외부 무언가 바꾸지 않는 함수
- 부작용 있는 함수
  - 외부 무언가 바꾸는 함수
```javascript
let a = 1;
let b = 'string';

// 부작용 없는 함수
// - 함수명 부합
function isReady(flag) {
  if (flag) {
    …
  }
  …
}

// 부작용 있는 함수
// - 함수명 부합 X
function checkPermission() {
  permissionCheck = true;
  a = 3;
  b = 'number';
  …

  // 옳바른 반환 값
  // return permissionCheck;

  // 잘못된 반환 값
  return {
    permission: permissionCheck,
    x: 5,
    y: true
  }
}
```
